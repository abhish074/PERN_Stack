DROP TABLE IF EXISTS CLIENT_PERSONAL;
DROP TABLE IF EXISTS CLIENT_FINANCIALS;

CREATE TABLE CLIENT_PERSONAL
(
client_id INTEGER,
name VARCHAR(30),
gender VARCHAR(2),
age INTEGER,
city VARCHAR(30),
length_at_residence INTEGER,
education NUMERIC,
PRIMARY KEY(client_id)
);


CREATE TABLE CLIENT_FINANCIALS
(
client_id INTEGER,
income INTEGER,
house_value INTEGER,
employment_years INTEGER,
employment_category INTEGER,
credit_card_debt INTEGER,
charitable_donations INTEGER,
gold_membership INTEGER,
FOREIGN KEY(client_id)
	  REFERENCES CLIENT_PERSONAL(client_id)
)
